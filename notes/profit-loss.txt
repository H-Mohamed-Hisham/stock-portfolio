// Lib
import prisma from "@/lib/prisma";

// Provider
import { getSessionUser } from "@/providers/auth/get-session-user";

export const GET = async (request: any) => {
  try {
    const sessionUser = await getSessionUser();

    if (!sessionUser || !sessionUser.userId) {
      return new Response("Unauthorised Access : User ID is required", {
        status: 401,
      });
    }

    const { userId } = sessionUser;

    const transactions = await prisma.stockTransaction.groupBy({
      by: ["stock_id", "transaction_type"],
      where: {
        user_id: userId,
      },
      _sum: {
        shares: true,
      },
    });

    // Initialize an object to store net shares for each stock
    const netSharesMap: any = {};

    transactions.forEach((transaction: any) => {
      const stockId = transaction.stock_id;
      const transactionType = transaction.transaction_type;
      const shares = transaction._sum.shares;

      // Initialize netShares for the stockId if it doesn't exist
      if (!netSharesMap[stockId]) {
        netSharesMap[stockId] = 0;
      }

      // Add or subtract shares based on the transaction type
      if (transactionType === "buy") {
        netSharesMap[stockId] += shares;
      } else if (transactionType === "sell") {
        netSharesMap[stockId] -= shares;
      }
    });

    const sharesNotHolding: any = Object.keys(netSharesMap)
      .filter((stockId) => netSharesMap[stockId] === 0)
      .map((stockId) => ({ stock_id: stockId }));

    const result: any = [];

    for (const item of sharesNotHolding) {
      // const totalShares = item._sum.shares;

      const test = await prisma.stockTransaction.groupBy({
        by: ["stock_id"],
        where: {
          user_id: userId,
          stock_id: item.stock_id,
          transaction_type: "buy",
        },
        _sum: {
          shares: true,
        },
      });

      console.log("test :: ", test);

      const buyTransactions = await prisma.stockTransaction.findMany({
        where: {
          user_id: userId,
          stock_id: item.stock_id,
          transaction_type: "buy",
        },
      });
      const totalInvested = buyTransactions.reduce(
        (acc: any, t) => acc + t.total,
        0
      );
      const sellTransactions = await prisma.stockTransaction.findMany({
        where: {
          user_id: userId,
          stock_id: item.stock_id,
          transaction_type: "sell",
        },
      });
      const totalReturns = sellTransactions.reduce(
        (acc: any, t) => acc + t.total,
        0
      );

      const stock: any = await prisma.stock.findUnique({
        where: {
          stock_id: item.stock_id,
        },
      });
      const stockSymbol = stock.stock_symbol;

      result.push({
        stockSymbol,
        // totalShares,
        totalInvested,
        totalReturns,
        // profitLossStatus,
        // profitLossAmount,
        // status,
      });
    }

    console.log("res :: ", result);

    return new Response(JSON.stringify(sharesNotHolding), {
      status: 200,
    });
  } catch (error: any) {
    console.log("err :: ", error);
    return new Response("Failed to fetch stock profit/loss data", {
      status: 500,
    });
  }
};
